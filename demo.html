UIKit Components - Complete Reference

1. Base Component - Foundation

Create Element & DOM Manipulation

const container = new Base();
container.createElement('div', 'my-class');
container.element.textContent = 'Hello';
document.body.appendChild(container.getDOMElement());

Styling

const box = new Base();
box.createElement('div');
box.setStyle('width', '100px');
box.setStyle('height', '100px');
box.setStyle('background', '#ff6b6b');
box.setStyle('borderRadius', '5px');

Classes & Attributes

const element = new Base();
element.createElement('div');
element.element.textContent = 'Hover me';
element.addClass('ui-button');
element.addClass('ui-button-primary');
element.setAttribute('data-demo', 'hover-effect');

Chainable API

const btn = new Base();
btn.createElement('button')
.setStyle('padding', '10px')
.setStyle('background', '#52c41a')
.setStyle('color', 'white')
.setStyle('border', 'none')
.setStyle('borderRadius', '4px')
.setStyle('cursor', 'pointer');
btn.element.textContent = 'Chainable';

Events

const btn = new Base();
btn.createElement('button');
btn.element.textContent = 'Click me';
btn.addClass('ui-button');
btn.on('click', () => alert('Clicked!'));

Append & Remove

const parent = new Base();
parent.createElement('div');

const child1 = new Base();
child1.createElement('p');
child1.element.textContent = 'Child 1';

const child2 = new Base();
child2.createElement('p');
child2.element.textContent = 'Child 2';

parent.append(child1).append(child2);

8. Form Component

Basic Form with Validation

const form = new Form();
form.addField('name', {
  label: 'Name *',
  required: true
});
form.addField('email', {
  label: 'Email *',
  type: 'email',
  required: true,
  validation: (v) => v.includes('@') ? null : 'Invalid email'
});

if (form.validate()) {
  const data = form.getData();
}

Form Methods

const form = new Form();
form.addField('username', { label: 'Username' });
form.setData({username: 'john'}); // Set data
form.getData(); // Get all data
form.getFieldValue('username');
form.setFieldValue('username', 'jane');
form.validate(); // Validate all fields
form.reset(); // Reset form
form.clearErrors(); // Clear error messages

9. Grid Component (Advanced Table)

Basic Grid with Data

const grid = new Grid();
grid.setHeaders([
  { key: 'name', label: 'Name', sortable: true },
  { key: 'age', label: 'Age', sortable: true },
  { key: 'city', label: 'City' }
]);

grid.setRows([
  { name: 'John', age: 25, city: 'NYC' },
  { name: 'Jane', age: 30, city: 'LA' },
  { name: 'Bob', age: 35, city: 'Chicago' }
]);

Grid Features

const grid = new Grid();
// Sorting: Click on header
// Filtering: Type in search box (real-time)
// Pagination: Auto 50 rows per page
// Multi-select: Checkboxes in first column

grid.on('rowselect', (e) => console.log(e.detail.row));
grid.on('selectionchange', (e) => console.log(e.detail.selected));

grid.getSelectedRows(); // Get selected rows
grid.sortBy('name', 'asc');
grid.setFilter('search text');
grid.addRow({name: 'Alice', age: 28});
grid.removeRow(0);
grid.updateRow(0, {age: 26});

10. Panel Component (Container)

Basic Panel

const panel = new Panel('My Panel');

const content = new Base();
content.createElement('p');
content.element.textContent = 'Panel content here';

panel.setContent(content.getDOMElement());

const btn = new Button('Action', () => alert('Clicked'));
panel.getFooter().appendChild(btn.getDOMElement());

Panel Structure

const panel = new Panel('Title');

// Three parts:
panel.getHeader(); // Panel header (title)
panel.setContent(el); // Content area
panel.getFooter(); // Footer for buttons

// Collapsible:
panel.collapse(); // Hide content
panel.expand(); // Show content
panel.toggleCollapse();

11. Dialog Component (Modal)

Basic Dialog

const dialog = new Dialog('Dialog Title');
dialog.setSize('medium'); // small, medium, large

const content = new Base();
content.createElement('p');
content.element.textContent = 'Dialog content';

dialog.setContent(content.getDOMElement());

const btn = new Button('Close', () => dialog.close());
dialog.getFooter().appendChild(btn.getDOMElement());

document.body.appendChild(dialog.getDOMElement());
dialog.show();

Static Dialog Methods

// Alert dialog
Dialog.alert('Title', 'Message');

// Confirm dialog
Dialog.confirm('Confirm?', () => {
  console.log('Confirmed!');
}, () => {
  console.log('Cancelled!');
});
